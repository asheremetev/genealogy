const p = dv.current();
if (p.дата_рождения && p.дата_смерти) {

  const d = dv.date(String(p.дата_смерти));

    dv.paragraph(`**Возраст:** ~${years} лет`);
  }
}

## 👨‍👩‍👧 Семья

**Отец:** `= this.отец`
**Мать:** `= this.мать`

### Браки / семьи

```dataview
TABLE WITHOUT ID
  file.link AS "Семья",
  дата_брака AS "Дата брака",
  место_брака AS "Место",
  статус_брака AS "Статус"
FROM "families"
WHERE муж = this.file.link OR жена = this.file.link
SORT дата_брака ASC
```

### Дети

```dataview
TABLE WITHOUT ID
  file.link AS "Ребёнок",
  пол AS "Пол",
  дата_рождения AS "Рождение",
  место_рождения AS "Место"
FROM "persons"
WHERE отец = this.file.link OR мать = this.file.link
SORT дата_рождения ASC
```

### Братья и сёстры

```dataviewjs
const p = dv.current();
const siblings = dv.pages('"persons"').where(s =>
  s.file.name !== p.file.name &&
  (
    (p.отец && s.отец && s.отец.path === p.отец.path) ||
    (p.мать && s.мать && s.мать.path === p.мать.path)
  )
);

const rows = siblings.map(s => {
  const bothFather = p.отец && s.отец && s.отец.path === p.отец.path;
  const bothMother = p.мать && s.мать && s.мать.path === p.мать.path;
  const type = (bothFather && bothMother) ? "полные" :
               bothFather ? "единокровные" : "единоутробные";
  return [s.file.link, s.пол, s.дата_рождения, type];
});

if (rows.length > 0) {
  dv.table(["Имя", "Пол", "Рождение", "Тип связи"], rows);
} else {
  dv.paragraph("_Нет данных_");
}
```
