const p = dv.current();
if (p.дата_рождения && p.дата_смерти) {
  const b = dv.date(String(p.дата_рождения));
  const d = dv.date(String(p.дата_смерти));
  if (b && d) {
    const years = Math.floor((d - b) / 31557600000);
    dv.paragraph(`**Прожил(а):** ~${years} лет`);
  }
} else if (p.дата_рождения && p.жив === "да") {
  const b = dv.date(String(p.дата_рождения));
  if (b) {
    const years = Math.floor((Date.now() - b) / 31557600000);
    dv.paragraph(`**Возраст:** ~${years} лет`);
  }
}

## 👨‍👩‍👧 Семья

**Отец:** `= this.отец`
**Мать:** `= this.мать`

### Браки / семьи

```dataview
TABLE WITHOUT ID
  file.link AS "Семья",
  дата_брака AS "Дата брака",
  место_брака AS "Место",
  статус_брака AS "Статус"
FROM "families"
WHERE муж = this.file.link OR жена = this.file.link
SORT дата_брака ASC
```

### Дети

```dataview
TABLE WITHOUT ID
  file.link AS "Ребёнок",
  пол AS "Пол",
  дата_рождения AS "Рождение",
  место_рождения AS "Место"
FROM "persons"
WHERE отец = this.file.link OR мать = this.file.link
SORT дата_рождения ASC
```

### Братья и сёстры

```dataview
TABLE WITHOUT ID
  file.link AS "Имя",
  пол AS "Пол",
  дата_рождения AS "Рождение"
FROM "persons"
WHERE (отец = this.отец OR мать = this.мать)
  AND file.name != this.file.name
  AND (отец OR мать)
SORT дата_рождения ASC
```
